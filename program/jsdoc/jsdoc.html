とかエラーが表示されたら、JSdoc.pmのファイルを開いて

790行目の


sub clear_nested {
    my ($doc, $declaration, $funcbody) = @_;
    if ($doc !~ /\@(constructor|class)/){
        return "$doc\n${declaration}{}\n";
    }
    return "$doc\n$declaration$funcbody";
}
